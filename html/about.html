class CourierTrack extends Component {
	flag = 0;
	render() {
		const { items,	coordinates, latrest, longrest,textaction  } = this.props
		let distance = [];
		if(!items)
			return(<View></View>)
		let len = items.length;
		let min;
		let minindex;
		let lowerLat;
		for(let a=0;a<len;a++){
			if(latrest<items[a].coordinates.latrest)
				lowerLat = latrest;
			else
				lowerLat = items[a].coordinates.latrest;
			//distance[a] = Math.sqrt(Math.pow((latrest-items[a].coordinates.latcour),2)+Math.pow((longrest-items[a].coordinates.longcour),2));
			distance[a] = 1000*Math.sqrt(Math.pow((Math.abs(latrest-items[a].coordinates.latcour)*111.16),2)+Math.pow((Math.abs(longrest-items[a].coordinates.longcour)*111.3*0.745),2));
			if(a==0)
			{
				min = distance[a];
				minindex = a;
			}
			if(distance[a]<min)
			{
				min = distance[a];
				minindex = a;
			}
		}
		if(this.flag==0 && minindex!=undefined){
			try{
				items[minindex].textaction(min);
				this.flag++;
			}
			catch
			{
				
			}
		}
		return (
			<View style={styles.wrapper}>
				{distance.map((item)=>
				<Text>{item}</Text>)}
				<Text>Minimal: {Math.min.apply(Math,distance)}, minimal index: {minindex}</Text>
			</View>
		)
	}
}
